Basic Schema
[TABLE] users
-	id (INT), PK, AUTO_INCREMENT(INSERT 시 이 COLUMN은 NULL로 넣으면 알아서 추가됩니다)
-	user_id (VARCHAR(15)), not NULL
-	person_name (VARCHAR(10)), not NULL
-	password_hash (VARCHAR(150)), not NULL

[TABLE] buildings
-	building_code (VARCHAR(6)), not NULL, PK
-	building_name (VARCHAR(80)), not NULL

[TABLE] location_categories
-	id (INT), not NULL, PK
-	type (VARCHAR(20)), not NULL

[TABLE] locations
-	id (INT), not NULL, PK
-	location_name (VARCHAR(30)), not NULL
-	building_number (VARCHAR(6)), not NULL, FK to BUILDING.building_number
-	categories_id (INT) , not NULL, FK to CATEGORY.category_number
-	available_begin_weekday (TIME)
-	available_end_weekday (TIME)
-	available_begin_weekend (TIME)
-	available_end_weekend (TIME)

[TABLE] slots
-	id (INT), not NULL , PK
-	slot_name (VARCHAR(50)) , not NULL
-	locations_id (INT) , not NULL, FK to LOCATION.location_number
-	slottypes_id (INT) , not NULL, FK to SLOTTYPES.slot_number
-	max_reserve_time (INT)
-	minimum_capacity (INT)

[TABLE] slottypes	몇인실인지(열람실일 경우도 '개인 좌석'으로 표시), 최대 수용 인원은 몇 명인지에 대한 정보
-	id (INT) , not NULL
-	slot_type (VARCHAR(30))
-	maximum_capacity (INT)

[TABLE] reservations
-	id (INT) , not NULL
-	slots_id (INT) , not NULL, FK to SLOT.slot_id
-	users_id (VARCHAR(15)) , not NULL, FK to USER.user_id
-	begin_date (DATETIME) , not NULL
-	end_date (DATETIME) , not NULL
-	num_people (INT) , not NULL
-	reservation_purpose (VARCHAR(50))

[TABLE] reviews
-	id (INT) , not NULL
-	user_id (VARCHAR(15)) , not NULL
-	locations_id (INT) , not NULL
-	like_score (INT)
-	crowded_score (INT)
-	comment (VARCHAR(300))
-	timestamp (DATETIME) , not NULL
